107  float getFactor() {
108  int last_idx = time_points.size()-1;
109  if (last_idx <=1) return 1.0;
110  float timespan_local_ms = time_points[last_idx].local_ms - time_points[0].local_ms;
111  float timespan_server_ms = time_points[last_idx].server_ms - time_points[0].server_ms;
112  if (timespan_local_ms == 0.0 || timespan_server_ms == 0.0) {
113  ESP_LOGE(TAG, "Could not determine clock differences");
114  return 1.0;
115  }
116  // if server time span is smaller then local, local runs faster and needs to be slowed down
117  float result_factor = timespan_server_ms / timespan_local_ms;
118  ESP_LOGI(TAG, "=> adjusting playback speed by factor: %f", result_factor);
119  return result_factor;
120  }
121 protected:
122  Vector<SnapTimePoints> time_points;
